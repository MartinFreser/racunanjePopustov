\begin{large}\textbf{Psevdokoda}\end{large} \vspace{10mm}

Razlago spremenljivk in programa najdete v poglavju Koraki programa. \\

\begin{algorithm}[H]
\caption{Psevdokoda programa}
 \KwData{$P_u$, $\mu_d$, $\sigma_d$, $stPop$}
 \KwResult{M, nmg // M-meja za avtoriziranje, nmg-namig avtorizantu}
	\textbf{Initialization of local variables:}\\
	s $|P_n|$ oznaèimo moè populacije $P_n$\;
	$m$ := prièakovano število popustov\;
	$P_s[]$ := polje vseh nakljuènih populacij velikosti $stPop$\;
	nmgArray[] := polje namigov za trenutni dan velikosti $stPop$\;
	MArray[] := polje mej za trenutni dan velikost $stPop$\;
  $\mu$=mean($P_u$)+$\mu_d$\;
	$\sigma$=std($P_u$)+$\sigma_d$\;
	$M_z:=\mu+\sigma$ // zaèetna meja\;
	$k:=1$ // koeficient za premikanje meje\;
	\For{i:=1 \textbf{to} $stPop$}{
		$P_s[i]$ := nakljucna normalno porazdeljena populacija velikosti $m$ z $\mu$ in $\sigma$\;
	}
	$P_r:=\emptyset$\;
	$P_n:=\emptyset$\;
	\line(1,0){50} \\
	\For{vsak dan}{
		\For{i:=1 \textbf{to} $stPop$}{
			\begin{larger}$k:=\frac{\mu \cdot m-\sum_{\tiny{p\in P_r}}{p}}{\sum_{\tiny{p\in P_s}}{p}}$\end{larger}\;
			$MArray[i]:=k\cdot M_z$ // nova trenutna meja\;
			\begin{larger}$h=\frac{\mu \cdot m-\sum_{\tiny{p\in P_r}}{p}-\sum_{\tiny{p \in P_s[i], p < M}}{p}}{\sum_{\tiny{p \in P_s[i], p \geq M}}{p}}$\end{larger}\;
			$nmgArray[i] := 1-h$\;
			$P_n$ := nove prejete vrednosti\;
			iz $P_s[i]$ odstrani $|P_n|$ nakljuènih vrednosti\;
			$P_r:=P_r+P_n$\;
			%\eIf{understand}{
			 %go to next section\;
			 %current section becomes this one\;
			 %}{
			 %go back to the beginning of current section\;
			}
			$M = \text{mediana} (MArray)$ //meja za ta dan\;
			$nmg = \text{mediana}(nmgArray)$ // namig za ta dan\;
	}
\end{algorithm}

\begin{large}\textbf{Koraki programa}\end{large} \vspace{10mm}


%\textbf{\textcolor{red}{Koraki programa:}}
\begin{enumerate}
	\item Izraèunamo $\sigma$ (standardni odklon) in $\mu$ (povpreèje) uène populacije $P_u$.
	\item Generiramo polje velikosti $stPop$ simuliranih populacij $P_s[]$. Posamezna populacija $P_s[i]$ je velikosti $m$, katero obièajno doloèimo na podlagi števila popustov v enakem obdobju leto nazaj in indeksa rasti.
	\item Realna populacija $P_r$ je na zaèetku prazna.
	\item Nova populacija $P_n$ je na zaèetku prazna. \\
	
	Dnevno ponavljamo naslednje korake:\\
	Ponavljamo za i := 1 do $stPop$:\\
	\item Izraèunamo $k$ iz enaèbe:
	$$\mu = \frac{\sum_{\tiny{p\in P_r}}{p}+k\cdot \sum_{\tiny{p\in P_s[i]}}{p}}{m},$$
	kjer je $\mu$ povpreèje od $P_u$, $P_r$ je realna populacija, $P_n$ nova populacija, $P_s[i]$ simulirana populacija in $m$ velikost simulirane populacije. \\
	Èe $k=1$, pomeni, da mejo ohranjamo oz. se ta ne spremeni.
	\item Za mejo $MArray[i]$ postavimo vrednost $k\cdot(\mu+\sigma)$.
	\item Izraèunamo $h$ iz enaèbe:
	$$\mu=\frac{\sum_{\tiny{p\in P_r}}{p}+\sum_{\tiny{p \in P_s, p < M}}{p}+h\cdot\sum_{\tiny{p \in P_s, p \geq M}}{p}}{m}$$
	\item Za namig $nmgArray[i]$ postavimo vrednost $1-h$
	\item Novo populacijo $P_n$ prepišemo z novimi prejetimi vrednostmi.
	\item Iz simulirane populacije $P_s[i]$ nakljuèno odstranimo toliko elementov oz. popustov, kolikor jih je v novi populaciji $P_n$.
	\item K realni populaciji $P_r$ dodamo novo populacijo $P_n$.
	\item Ko zakljuèimo z vsemi simuliranimi populacijami doloèimo namig $nmg =$mediana $(nmgArray)$ in mejo $M = \text{mediana}(MArray)$
\end{enumerate} \vspace{10mm}

Funkcija \emph{generirajModel}, ki je tudi glavna funkcija programa, ima naslednje parametre:
